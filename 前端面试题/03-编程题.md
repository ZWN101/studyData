### 数组的去重

* 双重for循环

  ```js
  let arr=[1,-1,2,8,9,11,2];
  function unique(arr){
      let len=arr.length;
      for(let i=0;i<len;i++){
          for(let j=i+1;j<len;j++){
              if(arr[i]==arr[j]){
                  arr.splice(j,1);//删除重复的数值
                  j--;
                  len--;
              }
          }
      }
      return arr;
  }
  let newArr=unique(arr);
  console.log(newArr)//1,-1,2,8,9,11
  ```

* 利用数组的indexOf

  ```js
  var arr =[1,-5,-4,0,-4,7,7,3];
  function unique(arr){
      let newArr=[];
      for(let i=0;i<arr.length;i++){
          //indexOf返回值为-1说明数组里没有该值
          if(newArr.indexOf(arr[i])==-1){
              newArr.push(arr[i])
          }
      }
      return newArr;
  }
  console.log(unique(arr));//1,-5,-4,0,7,3
  ```

* 利用数组的filter方法

  ```js
  var arr = [1,2,8,9,5,8,4,0,4];
  function unique(arr){
      return arr.filter((item,index)=>{
         return arr.indexOf(item)==index;//indexOf返回数据在数组中的第一个索引值,如果和当前的index值不一样，就说明有重复
     })
  }
  console.log(unique(arr));//1,2,8,9,5,4,0
  ```

### 数组的排序

* sort排序

  ```js

  //升序排序
  arr.sort(function(a,b){
      return a-b;

  ```

* 冒泡排序

  ```js
  let arr=[[6,1,2,4,3,5]];
  function bubbleSort(arr){
      for(let i=0;i<arr.length-1;i++){
          for(let j=0;j<arr.length-1-i;j++){
              if(arr[i]>arr[j+1]){
                  var temp=arr[i];
                  arr[i]=arr[j+1];
                  arr[j+1]=temp;
              }
          }
      }
  }
  ```

### 冒泡排序

```js
var arr=[2,1,5,3,10,2,9,8];
function bubblingSort(arr){
    let len=arr.length;
    //排序的趟数
    for(let i=0;i<len;i++){
        //比较的次数
        for(let j=1;j<len-i;j++){
            if(arr[j-1]>arr[j]){
                var temp=arr[j];
                arr[j]=arr[j-1];
                arr[j-1]=temp;
            }
        }
    }
    return arr;
}
//缺点 效率不高
```

### 快速排序

原理：

找到基准，设置左右两边的哨兵i和j，j先向左边移动，找到小于基准的数字，停下来，i向右移动，找到大于基准的数停下来，i和j对应的值进行交换，j继续向右移动，重复刚才的动作，直到i和j碰面，基准与碰面的值交换位置，然后以该位置作为分割点，左右两边重复刚才的动作

```js
var arr=[2,1,5,3,10,2,9,8];
function quickSort(arr,begin,end){
    if(begin>end){
        return;
    }
    var l=begin;
    var r=end;
    var temp=arr[begin];//基准数
    while(l<r){
        while(l<r && arr[r]>=temp){
            r--;
        }
        while(l<r && arr[l]<temp){
            l++;
        }
        //交换位置
        [arr[l],arr[r]]=[arr[r],arr[l]]
    }
    //交换基准与左右指针相遇的位置
    [arr[begin],arr[l]]=[arr[l],arr[begin]];
    quickSort(arr,begin,l-1);//左    
    quickSort(arr,l+1,end);//右
}
quickSort(arr, 0, 7);
console.log(arr)
```
